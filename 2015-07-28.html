tmux用のzshrc

Date 2015年07月28日(火) Tags tmux / zsh / linux

完全に個人用設定の .zshrc

# ansibleで遊ぶ用
export CLOUDSDK_PYTHON_SITEPACKAGES=1
export ANSIBLE_HOST_KEY_CHECKING=False

# zsh起動時に強制的にvirtualenvに入る
if [ -n "$VIRTUAL_ENV" ]; then
  . "$VIRTUAL_ENV/bin/activate"
else
  workon py27
fi

# tmuxでsshで他サーバのログイン時にウィンドウ名をホスト名にする
ssh_tmux() {
    ssh_cmd="ssh $@"
    tmux new-window -n "$*" "$ssh_cmd"
}

if [ $TERM = "screen" ] ; then
    tmux lsw
    if [ $? -eq 0 ] ; then
        alias ssh=ssh_tmux
    fi
fi
